17  package org.bitstrings.maven.plugins.dependencypath;
18  
19  import java.io.File;
20  import java.util.Arrays;
21  import java.util.Properties;
22  import java.util.Set;
23  
24  import org.apache.maven.artifact.Artifact;
25  import org.apache.maven.plugin.AbstractMojo;
26  import org.apache.maven.plugin.MojoExecutionException;
27  import org.apache.maven.plugin.MojoFailureException;
28  import org.apache.maven.project.MavenProject;
29  import org.codehaus.plexus.util.PathTool;
30  
31  /**
32   * Sets a property pointing to the artifact file for each selected project dependency.
33   * Each property name will have a base name in form of groupId:artifactId:type:[classifier][.relative][.suffix].
34   *
35   * @requiresDependencyResolution test
36   * @goal set
37   * @phase initialize
38   * @threadSafe
39   *
40   * @author Pino Silvaggio
41   *
42   */
43  public class DependencyPathMojo extends AbstractMojo
44  {
45      /**
46       * @parameter expression = "${project}"
47       * @required
48       * @readonly
49       *
50       * @since 1.0.0
51       */
52      private MavenProject project;
53  
54      /**
55       * The default property set.
56       * See <a href="propertyset-reference.html">PropertySet Reference</a>.
57       *
58       * @parameter
59       *
60       * @since 1.0.0
61       */
62      private PropertySet defaultPropertySet;
63  
64      /**
65       * The property sets.
66       * See <a href="propertyset-reference.html">PropertySet Reference</a>.
67       *
68       * @parameter
69       *
70       * @since 1.0.0
71       */
72      private PropertySet[] propertySets;
73  
74      public void execute() throws MojoExecutionException, MojoFailureException
75      {
76          if (defaultPropertySet == null)
77          {
78              defaultPropertySet = new PropertySet();
79          }
80  
81          if (propertySets == null)
82          {
83              propertySets = new PropertySet[] { new PropertySet() };
84          }
85  
86          if (defaultPropertySet.getAutoRelativeSuffix() == null)
87          {
88              defaultPropertySet.setAutoRelativeSuffix(true);
89          }
90  
91          if (defaultPropertySet.getTransitive() == null)
92          {
93              defaultPropertySet.setTransitive(true);
94          }
95  
96          if (getLog().isDebugEnabled())
97          {
98              getLog().debug("defaultPropertySet" + defaultPropertySet);
99              getLog().debug("propertySets" + Arrays.toString(propertySets));
100         }
101 
102         final Properties properties = project.getProperties();
103 
104         for (PropertySet propertySet : propertySets)
105         {
106             final Set<String> includes = propertySet.getIncludes();
107             final Set<String> excludes = propertySet.getExcludes();
108 
109             Boolean transitive = propertySet.getTransitive();
110 
111             if (transitive == null)
112             {
113                 transitive = defaultPropertySet.getTransitive();
114             }
115 
116             final Set<Artifact> artifacts =
117                             transitive
118                                     ? project.getArtifacts()
119                                     : project.getDependencyArtifacts();
120 
121             for (Artifact artifact : artifacts)
122             {
123                 final String dependencyConflictId = artifact.getDependencyConflictId();
124 
125                 if (((includes != null) && !includes.isEmpty()
126                                 && !includes.contains(dependencyConflictId))
127                         || ((excludes != null) && !excludes.isEmpty()
128                                         && excludes.contains(dependencyConflictId)))
129                 {
130                     continue;
131                 }
132 
133                 String key = dependencyConflictId;
134 
135                 File relativeTo = propertySet.getRelativeTo();
136 
137                 if (relativeTo == null)
138                 {
139                     relativeTo = defaultPropertySet.getRelativeTo();
140                 }
141 
142                 if (relativeTo != null)
143                 {
144                     Boolean autoRelativeSuffix = propertySet.getAutoRelativeSuffix();
145 
146                     if (autoRelativeSuffix == null)
147                     {
148                         autoRelativeSuffix = defaultPropertySet.getAutoRelativeSuffix();
149                     }
150 
151                     if (autoRelativeSuffix)
152                     {
153                         key += ".relative";
154                     }
155                 }
156 
157                 String suffix = propertySet.getSuffix();
158 
159                 if (suffix == null)
160                 {
161                     suffix = defaultPropertySet.getSuffix();
162                 }
163 
164                 if (suffix != null)
165                 {
166                     key += "." + suffix;
167                 }
168 
169                 properties.setProperty(key, getPath(artifact, relativeTo));
170 
171                 if (getLog().isDebugEnabled())
172                 {
173                     getLog().debug("Setting property " + key + "=" + properties.get(key));
174                 }
175             }
176         }
177     }
178 
179     protected String getPath(Artifact artifact, File relativeTo)
180     {
181         return
182                 relativeTo == null
183                         ? artifact.getFile().getAbsolutePath()
184                         : PathTool.getRelativeFilePath(relativeTo.getPath(), artifact.getFile().getAbsolutePath());
185     }
